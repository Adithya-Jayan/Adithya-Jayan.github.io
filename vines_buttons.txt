<script>
(function () {
  const config = [
    { id: 'lf0', tilt: -12, freq: 0.75, phase: 0.0,  amp: 2.5, page: 'Home',    color: '#3c6e4a', hov: '#5a9a6a', act: '#6cb858' },
    { id: 'lf1', tilt:   8, freq: 0.70, phase: 1.3,  amp: 2.0, page: 'About',   color: '#4a7840', hov: '#6aaa5c', act: '#84c050' },
    { id: 'lf2', tilt:  -5, freq: 0.80, phase: 2.1,  amp: 2.5, page: 'Work',    color: '#2d6050', hov: '#459870', act: '#54b888' },
    { id: 'lf3', tilt:  14, freq: 0.68, phase: 0.8,  amp: 2.0, page: 'Journal', color: '#4e6c32', hov: '#70984a', act: '#8eba52' },
    { id: 'lf4', tilt: -16, freq: 0.76, phase: 1.7,  amp: 2.5, page: 'Gallery', color: '#3a5e4a', hov: '#58886a', act: '#68ac7a' },
    { id: 'lf5', tilt:   5, freq: 0.72, phase: 2.8,  amp: 2.0, page: 'Contact', color: '#2d604a', hov: '#459868', act: '#52b870' },
  ];

  config.forEach(c => {
    c.el   = document.getElementById(c.id);
    c.body = c.el.querySelector('.leaf-body');
    c.isHovered = false;
    c.isActive  = false;
    c.hoverStart = 0;

    c.el.addEventListener('mouseenter', () => {
      c.isHovered = true;
      c.hoverStart = performance.now() / 1000;
      c.body.setAttribute('fill', c.hov);
    });

    c.el.addEventListener('mouseleave', () => {
      c.isHovered = false;
      c.body.setAttribute('fill', c.isActive ? c.act : c.color);
    });

    c.el.addEventListener('click', () => {
      config.forEach(x => {
        x.isActive = false;
        if (!x.isHovered) x.body.setAttribute('fill', x.color);
      });
      c.isActive = true;
      if (!c.isHovered) c.body.setAttribute('fill', c.act);

      const s = document.getElementById('vine-status');
      s.classList.remove('show');
      requestAnimationFrame(() => requestAnimationFrame(() => {
        s.textContent = '→ ' + c.page;
        s.classList.add('show');
      }));
    });
  });

  function tick() {
    const t = performance.now() / 1000;
    config.forEach(c => {
      let angle, tf;
      if (c.isHovered) {
        const ht = t - c.hoverStart;
        angle = c.tilt + Math.sin(ht * 7.5) * 7;
        tf = `rotate(${angle.toFixed(3)}) scale(1.07)`;
      } else {
        angle = c.tilt + Math.sin(t * c.freq + c.phase) * c.amp;
        tf = `rotate(${angle.toFixed(3)})`;
      }
      c.el.setAttribute('transform', tf);
    });
    requestAnimationFrame(tick);
  }
  tick();
})();
</script>
