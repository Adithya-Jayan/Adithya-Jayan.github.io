<script>
(function () {
  const NS = 'http://www.w3.org/2000/svg';
  const cfg = [
    { l:'Home',    dir:-1, x:72,  c:'#3c6e4a', hv:'#5a9a6a', tilt:-9,  f:.74, p:0.0, a:2.3 },
    { l:'About',   dir: 1, x:184, c:'#4a7840', hv:'#6aaa5c', tilt: 8,  f:.69, p:1.2, a:2.0 },
    { l:'Work',    dir:-1, x:314, c:'#2d6050', hv:'#459870', tilt:-5,  f:.78, p:2.0, a:2.4 },
    { l:'Journal', dir: 1, x:428, c:'#4e6c32', hv:'#70984a', tilt:13,  f:.67, p:0.7, a:2.0 },
    { l:'Gallery', dir:-1, x:534, c:'#3a5e4a', hv:'#58886a', tilt:-13, f:.75, p:1.6, a:2.3 },
    { l:'Contact', dir: 1, x:624, c:'#2d604a', hv:'#459868', tilt: 6,  f:.71, p:2.7, a:1.9 },
  ];
  let ai = 0;

  function mk(tag, a) {
    const e = document.createElementNS(NS, tag);
    if (a) Object.entries(a).forEach(([k, v]) => e.setAttribute(k, String(v)));
    return e;
  }

  function mkLeaf(dir, color, label) {
    const d = dir;
    const g = mk('g', { class: 'lv' });
    const body = mk('path', {
      d: `M 0,0 C 27,${d*22} 34,${d*68} 0,${d*100} C -34,${d*68} -27,${d*22} 0,0 Z`,
      fill: color, class: 'lb'
    });
    g.appendChild(body);
    g.appendChild(mk('path', { d:`M 0,${d*2} C -1,${d*35} 0,${d*70} 0,${d*96}`, fill:'none', stroke:'#1a3a20', 'stroke-width':'0.9', opacity:'0.36' }));
    g.appendChild(mk('path', { d:`M -1,${d*36} C -13,${d*47} -17,${d*51}`, fill:'none', stroke:'#1a3a20', 'stroke-width':'0.6', opacity:'0.27' }));
    g.appendChild(mk('path', { d:`M 0,${d*56} C 12,${d*68} 15,${d*73}`, fill:'none', stroke:'#1a3a20', 'stroke-width':'0.6', opacity:'0.27' }));
    g.appendChild(mk('path', { d:`M -1,${d*21} C -9,${d*28} -12,${d*31}`, fill:'none', stroke:'#1a3a20', 'stroke-width':'0.5', opacity:'0.21' }));
    const t = mk('text', { x:'0', y:String(d*48), 'text-anchor':'middle', 'dominant-baseline':'central' });
    t.style.cssText = 'font-family:var(--font-serif);font-style:italic;font-size:12px;fill:rgba(236,226,205,0.93)';
    t.textContent = label;
    g.appendChild(t);
    return g;
  }

  function mkFlower(dir, label) {
    const d = dir;
    const g = mk('g', { class: 'fv off' });

    // Stem (slightly curved, more botanical)
    const stemPath = mk('path', {
      d: `M 0,${d*4} C ${d*-2},${d*32} ${d*2},${d*52} 0,${d*70}`,
      fill:'none', stroke:'#5c7c38', 'stroke-width':'2.1', 'stroke-linecap':'round'
    });
    g.appendChild(stemPath);

    // Sepals (small green leaves at stem-head junction)
    g.appendChild(mk('path', { d:`M 0,${d*54} C -8,${d*61} -13,${d*67} -5,${d*61} Z`, fill:'#4a7040', opacity:'0.75' }));
    g.appendChild(mk('path', { d:`M 0,${d*54} C  8,${d*61}  13,${d*67}  5,${d*61} Z`, fill:'#4a7040', opacity:'0.75' }));

    // Flower head group
    const h = mk('g', { transform:`translate(0,${d*72})` });

    // Outer petals (5) — blush pink
    for (let i = 0; i < 5; i++) {
      h.appendChild(mk('ellipse', { cx:'0', cy:'-21', rx:'10', ry:'18', fill:'#f6cad8', opacity:'0.87', transform:`rotate(${i*72})` }));
    }
    // Middle petals (5, offset 36°) — deeper rose
    for (let i = 0; i < 5; i++) {
      h.appendChild(mk('ellipse', { cx:'0', cy:'-13', rx:'6.5', ry:'11', fill:'#e898ba', opacity:'0.90', transform:`rotate(${36 + i*72})` }));
    }
    // Sheen on outer petals (subtle highlight)
    for (let i = 0; i < 5; i++) {
      h.appendChild(mk('ellipse', { cx:'-1.5', cy:'-25', rx:'3.5', ry:'7', fill:'#fce8f0', opacity:'0.38', transform:`rotate(${i*72})` }));
    }
    // Golden center
    h.appendChild(mk('circle', { r:'13', fill:'#fcd060' }));
    h.appendChild(mk('circle', { r:'8',  fill:'#e8a820' }));
    h.appendChild(mk('circle', { r:'4',  fill:'#d09010', opacity:'0.7' }));
    // Stamen dots around center
    for (let i = 0; i < 8; i++) {
      const ang = i * 45 * Math.PI / 180;
      h.appendChild(mk('circle', {
        cx: (Math.cos(ang) * 5.5).toFixed(1),
        cy: (Math.sin(ang) * 5.5).toFixed(1),
        r: '1.5', fill: '#9c6c10'
      }));
    }

    g.appendChild(h);

    // Label — clear of petals (petal tip ~37px from head center, head at d*72, so tip at d*109)
    // Label at d*126 gives 17px clearance from petal tip
    const t = mk('text', { x:'0', y:String(d * 126), 'text-anchor':'middle', 'dominant-baseline':'central' });
    t.style.cssText = 'font-family:var(--font-serif);font-style:italic;font-size:12px;fill:var(--color-text-primary)';
    t.textContent = label;
    g.appendChild(t);

    return g;
  }

  const lgs = [], fgs = [], ogs = [];

  cfg.forEach((item, i) => {
    const og = document.getElementById(`n${i}`);
    ogs.push(og);
    const lg = mkLeaf(item.dir, item.c, item.l);
    const fg = mkFlower(item.dir, item.l);
    og.appendChild(lg);
    og.appendChild(fg);
    lgs.push(lg);
    fgs.push(fg);

    og.addEventListener('mouseenter', () => {
      item.hovered = true;
      item.hoverStart = performance.now() / 1000;
      if (i !== ai) lg.querySelector('.lb').setAttribute('fill', item.hv);
    });
    og.addEventListener('mouseleave', () => {
      item.hovered = false;
      if (i !== ai) lg.querySelector('.lb').setAttribute('fill', item.c);
    });
    og.addEventListener('click', () => {
      if (i === ai) return;
      // Wilt old flower → reveal leaf
      fgs[ai].classList.add('off');
      lgs[ai].classList.remove('off');
      // Bloom new flower → hide leaf
      lgs[i].classList.add('off');
      fgs[i].classList.remove('off');
      ai = i;
      const vsEl = document.getElementById('vs');
      vsEl.classList.remove('on');
      requestAnimationFrame(() => requestAnimationFrame(() => {
        vsEl.textContent = '→ ' + item.l;
        vsEl.classList.add('on');
      }));
    });
  });

  // Bootstrap: bloom Home on load
  requestAnimationFrame(() => requestAnimationFrame(() => {
    lgs[0].classList.add('off');
    fgs[0].classList.remove('off');
  }));

  // Sway loop
  function tick() {
    const t = performance.now() / 1000;
    cfg.forEach((item, i) => {
      let ang;
      if (item.hovered && i !== ai) {
        const ht = t - item.hoverStart;
        ang = item.tilt + Math.sin(ht * 7.2) * 8;
      } else if (i === ai) {
        // Flower: gentle graceful sway
        ang = item.tilt * 0.32 + Math.sin(t * item.f * 0.6 + item.p) * 0.85;
      } else {
        ang = item.tilt + Math.sin(t * item.f + item.p) * item.a;
      }
      ogs[i].setAttribute('transform', `translate(${item.x},161) rotate(${ang.toFixed(3)})`);
    });
    requestAnimationFrame(tick);
  }
  tick();
})();
</script>
